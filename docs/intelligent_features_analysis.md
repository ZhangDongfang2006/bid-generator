class DataLearner:
    """æ•°æ®å­¦ä¹ æ¨¡å—"""
    
    def __init__(self):
        self.model = RandomForestClassifier(n_estimators=100)
        self.vectorizer = TfidfVectorizer(max_features=100)
        self.is_trained = False
    
    def train_from_history(self, history_data):
        """ä»å†å²æ•°æ®ä¸­è®­ç»ƒæ¨¡å‹"""
        # æå–ç‰¹å¾å’Œæ ‡ç­¾

        # è®­ç»ƒæ¨¡å‹
        self.model.fit(features, labels)
        self.is_trained = True
    
    def predict_success_rate(self, requirement):
        """é¢„æµ‹ä¸­æ ‡ç‡"""
        if not self.is_trained:
            return 0.5  # æœªè®­ç»ƒæ—¶è¿”å› 0.5
        
        features = self.vectorizer.transform([requirement])
        prediction = self.model.predict(features)[0]
        return prediction
    
    def optimize_matching(self, requirements, qualifications):
        """ä¼˜åŒ–åŒ¹é…ç®—æ³•"""

            return qualifications  # æœªè®­ç»ƒæ—¶è¿”å›åŸå§‹ç»“æœ
        
        # å¯¹æ¯ä¸ªèµ„è´¨è®¡ç®—æˆåŠŸæ¦‚ç‡

                q_level_lower = q['level'].lower()
                
                # è®¡ç®—ç›¸å…³æ€§åˆ†æ•°

                    score += 0.5

                    score += 0.3
                
                relevance += score
            
            # æ ¹æ®æˆåŠŸæ¦‚ç‡æ’åº
            optimized_results.append({
                'qualification': q,
                'success_probability': self.predict_success_rate(q['name']),

        # å…ˆæŒ‰æˆåŠŸæ¦‚ç‡æ’åºï¼Œå†æŒ‰ç›¸å…³æ€§æ’åº
        optimized_results.sort(key=lambda x: (-x['success_probability'], -x['relevance_score']))
